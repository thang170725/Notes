Change items
Bạn có thể thay đổi giá trị của một mục cụ thể bằng cách tham chiếu đến khóa của nó.
def main():
    di = {
        "name": "John",
        "age": 30
    }
    di["age"] = 10
    print(di)
main()
{'name': 'John', 'age': 10}

Add items
Để thêm cặp key – value vào dictionary.
thisdict =	{
  "brand": "Ford",
  "model": "Mustang",
  "year": 1964
}
thisdict["color"] = "red"
print(thisdict)
{'brand': 'Ford', 'model': 'Mustang', 'year': 1964, 'color': 'red'}
thisdict = {
  "brand": "Ford",
  "model": "Mustang",
  "year": 1964
}
thisdict.update({"color": "red"})

print(thisdict)
{'brand': 'Ford', 'model': 'Mustang', 'year': 1964, 'color': 'red'}
